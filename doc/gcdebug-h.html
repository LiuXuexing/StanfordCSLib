/*
 * File: gcdebug.h
 * ---------------
 * This file specifies various debugging functions for the
 * garbage-collecting allocator.
 */

#ifndef _gcdebug_h
#define _gcdebug_h

#include "cslib.h"

/*
 * Type: gcStateT
 * --------------
 * This enumeration type holds the possible states of blocks being managed
 * by the garbage collector:
 *
 *   GCInUseBlock is in use  GCOnFreeListBlock is on  use  GCIUnknown
 *     Block is unknown to the allocator
 */
typedef enum { GCInUse, GCOnFreeList, GCUnknown } gcStateT;

/*
 * Function: gcTrace
 * Usage: gcTrace(flag);
 * ---------------------
 * Enables a trace of garbage collection events, which is printed to
 * stderr.
 */

void gcTrace(bool flag);

/*
 * Function: gcCheckBlock
 * Usage: state = gcCheckBlock(ptr);
 * ---------------------------------
 * Returns the state of the block at the specified address.
 */

gcStateT gcCheckBlock(void *ptr);

#endif
